** Diff
Change diff tool to [[https://github.com/jeffkaufman/icdiff][icdiff]]:
#+BEGIN_SRC conf
[diff]
  tool = icdiff
[difftool]
  prompt = false
[difftool "icdiff"]
  cmd = /usr/local/bin/icdiff --line-numbers $LOCAL $REMOTE
#+END_SRC
